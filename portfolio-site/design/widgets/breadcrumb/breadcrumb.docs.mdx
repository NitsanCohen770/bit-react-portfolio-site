import { Breadcrumb } from './breadcrumb';
import { lightTheme } from '@showoff/design.theme.light-theme';
import { ThemeProvider } from '@showoff/design.theme.theme-provider';

A component that renders a breadcrumb to display the stream of routes.

### Component usage

It is encouraged to wrap this with a `ThemeProvider` to enforce theming.

```js
<ThemeProvider theme={lightTheme()}>
  <Breadcrumb
    typographyProps={{
      fontSize: 30,
    }}
    crumbs={[
      { crumb: 'Home', link: '/' },
      { crumb: 'Components' },
      { crumb: 'Breadcrumb', link: '/component/breadcrumb' },
    ]}
  />
</ThemeProvider>
```
