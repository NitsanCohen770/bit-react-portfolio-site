import { Card } from './card';
import { darkTheme } from '@enlear/bit-react-portfolio-site-design-system.theme.dark-theme';
import { ThemeProvider } from '@enlear/bit-react-portfolio-site-design-system.theme.theme-provider';

A component that renders a card.

### Component usage

Use with a `ThemeProvider`

```js
<ThemeProvider theme={darkTheme()}>
  <Card
    title={'Project Name'}
    tags={['React', 'VSCode', 'Bit', 'Material-UI']}
    bannerImage={
      'https://images.unsplash.com/photo-1669798158874-9c05aa89d71b?ixlib=rb-4.0.3&ixid=MnwxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8&auto=format&fit=crop&w=1770&q=80'
    }
    subtitle={'This is a sample project description'}
  />
</ThemeProvider>
```
